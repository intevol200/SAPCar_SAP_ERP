MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SAVE'.
      PERFORM create_customer_info.
  ENDCASE.
ENDMODULE.

MODULE exit INPUT.
  CASE ok_code.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
  ENDCASE.

ENDMODULE.

MODULE user_command_0200 INPUT.

  ENDCASE.

ENDMODULE.

MODULE user_command_0300 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'DELETE'.
      CLEAR gv_subrc.
      PERFORM popup USING 'DELETE'
                          'Do you want to delete data?'
                  CHANGING gv_subrc.
      IF gv_subrc <> 0.
        RETURN.
      ENDIF.
      PERFORM delete_customer_info.
  ENDCASE.
ENDMODULE.
